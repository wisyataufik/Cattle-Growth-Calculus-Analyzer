    <script>
        // Core Biological Parameters (Defaults matching Malaysian KK Cattle feedlot and intensive system guidelines)
        const birthWeight = 15; // kg, average birth weight of KK calves
        let K = 350; // kg, Asymptotic mature weight 
        let b = 0.22; // Growth coefficient (months^-1)
        let a = (K - birthWeight) / birthWeight; // Mathematically ensures W(0) = birthWeight

        // Keep track of user state
        let currentMonth = 12; // Initial state point for analysis
        let chartInstance = null;

        // Dom Elements references
        const sliderK = document.getElementById('slider-k');
        const sliderB = document.getElementById('slider-b');
        const sliderT = document.getElementById('slider-t');
        const valK = document.getElementById('val-k');
        const valB = document.getElementById('val-b');
        const valTBadge = document.getElementById('val-t-badge');

        const calcMonth = document.getElementById('calc-month');
        const calcTVal = document.getElementById('calc-t-val');
        const calcWtResult = document.getElementById('calc-wt-result');
        const calcDwtResult = document.getElementById('calc-dwt-result');
        const calcAdgResult = document.getElementById('calc-adg-result');
        const laymanExplanation = document.getElementById('layman-explanation');

        // Mathematical calculation: Weight W(t)
        function calculateWeight(t, tempK = K, tempB = b) {
            const tempA = (tempK - birthWeight) / birthWeight;
            return tempK / (1 + tempA * Math.exp(-tempB * t));
        }

        // Mathematical calculation: Derivative W'(t) in kg/month
        function calculateGrowthRate(t, tempK = K, tempB = b) {
            const tempA = (tempK - birthWeight) / birthWeight;
            const numerator = tempK * tempA * tempB * Math.exp(-tempB * t);
            const denominator = Math.pow(1 + tempA * Math.exp(-tempB * t), 2);
            return numerator / denominator;
        }

        // Biological Milestones and Expert Analysis for Layman
        function getBioscienceInsight(t, wt, dwt, adg) {
            let desc = "";
            const infPoint = Math.log((K - birthWeight) / birthWeight) / b; // Point of inflection where growth rate peaks

            if (t < 3) {
                desc = `<strong>Pre-Weaning Colostrum & Liquid Diet Stage (Phase I)</strong>:<br/> At ${t.toFixed(1)} months, the KK calf is heavily reliant on maternal milk. Cellular division (hyperplasia) is intense. The high relative growth rate is driven by liquid nutrients. Ensure correct creep feed supplementation to support early rumen development in tropical conditions.`;
            } else if (t >= 3 && t < 7) {
                desc = `<strong>Weaning & Rumen Transition Window</strong>:<br/> Weaning typically happens around 6 months in pasture-intensive setups in Malaysia. The drop or plateau in growth rate at this age represents post-weaning shock. Standard management requires introducing quality forage like Napier grass or Palm Kernel Cake (PKC) pellets to mitigate rumen adaptation issues.`;
            } else if (t >= 7 && t < 14) {
                // Check if near inflection point
                const diffFromPeak = Math.abs(t - infPoint);
                if (diffFromPeak < 1.5) {
                    desc = `<strong>Peak Metabolic Phase (Physiological Puberty) [INFLECTION POINT]</strong>:<br/> You are looking at the maximum growth speed around the inflection point (${infPoint.toFixed(1)} months)! This is the most feed-efficient period of the animal's lifetime. Feedlot operational margin is highest here. Maximize protein/energy ratio in the feed ration to achieve maximum carcass yield later.`;
                } else {
                    desc = `<strong>Metabolic Acceleration/Grower Stage</strong>:<br/> At ${t.toFixed(1)} months, skeletal and muscular growth is fully accelerating. High efficiency of feed conversion (FCR) is observed. Ideal feedlot backgrounding stage. Keep the animals healthy and stress-free under the warm, humid Malaysian climate.`;
                }
            } else if (t >= 14 && t < 20) {
                desc = `<strong>Fattening Phase (Metabolic Deceleration)</strong>:<br/> At ${t.toFixed(1)} months, the derivative growth rate starts decreasing. Muscle synthesis slows, and adipose tissue (marbling) deposition is beginning. Cattle are highly resilient but require denser energy feeds to sustain fat growth. This is the finishing feedlot stage.`;
            } else {
                desc = `<strong>Physiological Maturity & Optimal Slaughter Age</strong>:<br/> The growth rate has dropped to a flat line (${dwt.toFixed(2)} kg/month). Keeping the KK cattle past ${t.toFixed(1)} months increases feed waste, as most nutrients are used just for basic body maintenance. Plan for immediate marketing/slaughter to optimize profitability (typically matching high demand cycles like Hari Raya Aidiladha in Malaysia).`;
            }
            return desc;
        }

        // Update the textual displays and calculations
        function updateCalculatedResults(t) {
            currentMonth = t;
            
            // Perform Math
            const wt = calculateWeight(t);
            const dwt = calculateGrowthRate(t);
            // Average Daily Gain in grams/day (kg/month transformed: divided by 30.4 days/mo * 1000g/kg)
            const adg = (dwt / 30.41) * 1000;

            // Update UI Numbers
            calcMonth.innerText = t.toFixed(1);
            calcTVal.innerText = t.toFixed(1);
            calcWtResult.innerText = wt.toFixed(1);
            calcDwtResult.innerText = dwt.toFixed(2);
            calcAdgResult.innerText = Math.round(adg);

            // Layman bioscience advice
            laymanExplanation.innerHTML = getBioscienceInsight(t, wt, dwt, adg);

            // Synchronize Timeline sliders if updated from clicking on the graph
            sliderT.value = t;
            valTBadge.innerText = `${t.toFixed(1)} Months`;
        }

        // Dynamically update the visual formula values of a and b
        function updateFormulaLaTeX() {
            const currentA = ((K - birthWeight) / birthWeight).toFixed(2);
            const wtFormula = `W(t) = \\frac{${K}}{1 + ${currentA} e^{-${b.toFixed(2)} t}}`;
            const dwtFormula = `W'(t) = \\frac{${K} \\cdot ${currentA} \\cdot ${b.toFixed(2)} e^{-${b.toFixed(2)} t}}{(1 + ${currentA} e^{-${b.toFixed(2)} t})^2}`;

            document.getElementById('latex-formula-wt').innerHTML = `$$${wtFormula}$$`;
            document.getElementById('latex-formula-dwt').innerHTML = `$$${dwtFormula}$$`;

            // Tell MathJax to re-render the formulas
            if (window.MathJax && window.MathJax.typesetPromise) {
                window.MathJax.typesetPromise();
            }
        }

        // Render the Chart.js visualization
        function buildChart() {
            const ctx = document.getElementById('growthChart').getContext('2d');

            // Generate growth data arrays for age 0 to 24 months
            const labels = [];
            const weightData = [];
            const rateData = [];

            for (let i = 0; i <= 24; i += 0.5) {
                labels.push(i);
                weightData.push(calculateWeight(i));
                rateData.push(calculateGrowthRate(i));
            }

            // If chart exists already, destroy it to redraw
            if (chartInstance) {
                chartInstance.destroy();
            }

            chartInstance = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [
                        {
                            label: 'Weight W(t) (kg)',
                            data: weightData,
                            borderColor: '#3b82f6', // Premium Blue
                            backgroundColor: 'rgba(59, 130, 246, 0.08)',
                            yAxisID: 'yWeight',
                            borderWidth: 4,
                            pointRadius: 0,
                            pointHoverRadius: 8,
                            pointHitRadius: 10,
                            fill: true,
                            tension: 0.2
                        },
                        {
                            label: 'Growth Rate W\'(t) (kg/month)',
                            data: rateData,
                            borderColor: '#10b981', // Emerald Green
                            backgroundColor: 'transparent',
                            yAxisID: 'yRate',
                            borderWidth: 3,
                            borderDash: [5, 5],
                            pointRadius: 0,
                            pointHoverRadius: 8,
                            pointHitRadius: 10,
                            tension: 0.2
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            display: false // We have a custom designed legend
                        },
                        tooltip: {
                            enabled: true,
                            mode: 'index',
                            intersect: false,
                            callbacks: {
                                label: function(context) {
                                    let label = context.dataset.label || '';
                                    if (label) {
                                        label += ': ';
                                    }
                                    if (context.parsed.y !== null) {
                                        label += context.parsed.y.toFixed(2);
                                    }
                                    return label;
                                }
                            }
                        }
                    },
                    // Click handler to get selected points directly from chart clicking
                    onClick: (e) => {
                        const points = chartInstance.getElementsAtEventForMode(e, 'index', { intersect: false }, true);
                        if (points.length > 0) {
                            const index = points[0].index;
                            const monthVal = labels[index];
                            updateCalculatedResults(monthVal);
                            drawVerticalIndicatorLine(monthVal);
                        }
                    },
                    scales: {
                        x: {
                            title: {
                                display: true,
                                text: 'Age of KK Calf (Months)',

                            const xValue = chart.scales.x.getPixelForValue(currentMonth);
                            const topY = chart.scales.yWeight.top;
                            const bottomY = chart.scales.yWeight.bottom;

                            ctx.save();
                            ctx.beginPath();
                            ctx.moveTo(xValue, topY);
                            ctx.lineTo(xValue, bottomY);
                            ctx.lineWidth = 2;
                            ctx.strokeStyle = '#f59e0b'; // Amber tracker line
                            ctx.setLineDash([4, 4]);
                            ctx.stroke();
                            ctx.restore();
